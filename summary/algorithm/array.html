            win[c] = (win[c] || 0) + 1

        }

            if(valid === count) res.push(left)

            left++

                if(win[c] === need[c]) valid--
                win[c]--
            }
        }
    }
    return res
};

var corpFlightBookings = function(bookings, n) {
    const diff = Array(n).fill(0)
    for(let [i, j, val] of bookings) {
        diff[i-1] += val

    }
